import { Inspection } from "../models/Inspection.js";import { pool } from "../config/db.js";import { notifyImporterByDeclaration } from "../utils/notify.js";import { scoreRisk } from "../utils/risk.js";export const getInspections = async (req, res) => {  try {    // Include risk scoring with context (shipment + latest performance)    const rows = await pool.query(`      SELECT i.*, d.declaration_no, s.cif_value_usd, s.goods_type, s.origin_country, s.importer_id      FROM inspections i      JOIN declarations d ON i.declaration_id = d.declaration_id      JOIN shipments s ON d.shipment_id = s.shipment_id      ORDER BY i.inspection_date DESC;    `);    const result = [];    for (const r of rows.rows) {      let perf = null;      try {        const p = await pool.query(          'SELECT * FROM performance WHERE importer_id=$1 ORDER BY created_at DESC LIMIT 1',          [r.importer_id]        );        perf = p.rowCount ? p.rows[0] : null;      } catch {}      const risk = scoreRisk({ shipment: r, performance: perf });      result.push({ ...r, risk_score: risk.score, risk_reasons: risk.reasons });    }    res.json(result);  } catch (err) {    res.status(500).json({ message: err.message });  }};export const createInspection = async (req, res) => {  try {    const payload = { ...req.body };    // Auto-generate release_reference if not provided    if (!payload.release_reference) {      const d = new Date();      const y = d.getFullYear();      const m = String(d.getMonth() + 1).padStart(2, '0');      const day = String(d.getDate()).padStart(2, '0');      const rand = Math.floor(Math.random() * 10000).toString().padStart(4, '0');      payload.release_reference = `REL-${y}${m}${day}-${rand}`;    }    // Risk scoring if not provided    if (!payload.risk_channel) {      // Find shipment context via declaration      let shipment = null, perf = null;      try {        const d = await pool.query('SELECT shipment_id FROM declarations WHERE declaration_id=$1', [payload.declaration_id]);        if (d.rowCount) {          const s = await pool.query('SELECT * FROM shipments WHERE shipment_id=$1', [d.rows[0].shipment_id]);          shipment = s.rowCount ? s.rows[0] : null;          if (shipment?.importer_id) {            const p = await pool.query(              'SELECT * FROM performance WHERE importer_id=$1 ORDER BY created_at DESC LIMIT 1',              [shipment.importer_id]            );            perf = p.rowCount ? p.rows[0] : null;          }        }      } catch {}      const risk = scoreRisk({ shipment, performance: perf });      payload.risk_channel = risk.channel;    }    const inspection = await Inspection.create(payload);    try { await notifyImporterByDeclaration({ declaration_id: payload.declaration_id, title: "Inspection Scheduled", message: `Your declaration has been scheduled for inspection. Reference: ${inspection.release_reference}` }); } catch {}    res.status(201).json(inspection);  } catch (err) {    res.status(500).json({ message: err.message });  }};
